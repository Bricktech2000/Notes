### operations and properties

see [[math-notation]]

let $\mathbb R^n (u, v, w) \dashv \mathbb Z n$

let $\mathbb R (c1, c2)$

$u \cdot 0 = 0$

$u \cdot \circ u = 0$

$(u \cdot v) \cdot w = u \cdot (v \cdot w)$

$u \cdot v = v \cdot u$

$c \mid u \cdot v = cu \cdot cv$

$c_1 \cdot c_2 \mid u = c_1u \cdot c_2u$

$c_1 c_2 \mid u = c_1 \mid c_2u$

$1u = u$
